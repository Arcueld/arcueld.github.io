2025/5/12 kernel

常规MmMapIoSpace

EXTERN_C NTSTATUS DriverEntry(PDRIVER_OBJECT DriverObject, PUNICODE_STRING RegistryPath){

    DriverObject->DriverUnload = (PDRIVER_UNLOAD)UnloadDriver;
    // 001ad000
    PHYSICAL_ADDRESS phyAddr = { 0 };
    phyAddr.QuadPart = 0x001ad000;
    ULONG64 lpMem = (ULONG64)MmMapIoSpace(phyAddr, PAGE_SIZE, MmNonCached);
    DbgPrintEx(77, 0, "lpMem: %llx\n", lpMem);




    return STATUS_SUCCESS;
}

win10下映射不了 可以映射常规的物理页 但是不能映射cr3相关 直接返回0

正常物理地址是可以映射的

    PHYSICAL_ADDRESS phyAddr = MmGetPhysicalAddress(UnloadDriver);


    // PHYSICAL_ADDRESS phyAddr = { 0 };
    // phyAddr.QuadPart = 0x000003c50000;

    ULONG64 lpMem = (ULONG64)MmMapIoSpace(phyAddr, PAGE_SIZE, MmNonCached);
    DbgPrintEx(77, 0, "lpMem: %llx\n", lpMem);

那么这里就要用到另一种映射的姿势

通过设备PhysicalMemory

这是一个Section 可以理解为就是物理内存

我们直接通过ZwMapViewOfSection映射

#include <ntifs.h>    

#define CHECK_STATUS_AND_BREAK(s)                                      \
    if (!NT_SUCCESS(s)) {                                              \
        DbgPrintEx(77, 0, "[%s:%d] status: 0x%llx\n",                   \
                   __FUNCTION__, __LINE__, (ULONGLONG)(s));            \
        break;                                                         \
    }


NTSTATUS UnloadDriver(PDRIVER_OBJECT DriverObject){
    DbgPrintEx(77, 0, "Unload!\n");

    return STATUS_SUCCESS;
}

EXTERN_C NTSTATUS DriverEntry(PDRIVER_OBJECT DriverObject, PUNICODE_STRING RegistryPath){

    DriverObject->DriverUnload = (PDRIVER_UNLOAD)UnloadDriver;
    // 001ad000
   
    PHYSICAL_ADDRESS phyAddr = { 0 };
    phyAddr.QuadPart = 0x001ad000;

    UNICODE_STRING uniDeviceStr = { 0 };
    RtlInitUnicodeString(&uniDeviceStr, L"\\DEVICE\\PhysicalMemory");
    HANDLE hSection = NULL;
    OBJECT_ATTRIBUTES obj = { 0 };
    InitializeObjectAttributes(&obj, &uniDeviceStr, OBJ_CASE_INSENSITIVE, NULL, NULL);
    do {
        NTSTATUS status = ZwOpenSection(&hSection, SECTION_ALL_ACCESS, &obj);
        
        CHECK_STATUS_AND_BREAK(status);

        PVOID lpMem = NULL;
        LARGE_INTEGER offset = { 0 };
        offset.QuadPart = 0x001ad000;
        SIZE_T size = PAGE_SIZE;
        status = ZwMapViewOfSection(hSection, NtCurrentProcess(), &lpMem, NULL, PAGE_SIZE, &offset, &size, ViewUnmap, MEM_TOP_DOWN, PAGE_EXECUTE_READWRITE);
        
        CHECK_STATUS_AND_BREAK(status);

        
        DbgPrintEx(77, 0, "lpMem: %llx\n", lpMem);



    } while (false);





    return STATUS_SUCCESS;
}
